 72:     /**
 73:      * Cache data by given key.
 74:      * 
 75:      * For consistency, the key will go through sha1() before it is saved.
 76:      * 
 77:      * This method implements {@link MailWizzApi_Cache_Abstract::set()}.
 78:      * 
 79:      * @param string $key
 80:      * @param mixed $value
 81:      * @return bool
 82:      */
 83:     public function set($key, $value)
 84:     {
 85:         $value = serialize($value);
 86:         if ($exists = $this->get($key)) {
 87:             if ($value === serialize($exists)) {
 88:                 return true;
 89:             }
 90:             $key = sha1($key);
 91:             $con = $this->getConnection();
 92:             $sth = $con->prepare('UPDATE `'.$this->getTableName().'` SET `value` = :v WHERE `key` = :k');
 93:             return $sth->execute(array(':v' => $value, ':k' => $key));
 94:         }
 95:         $key = sha1($key);
 96:         $con = $this->getConnection();
 97:         $sth = $con->prepare('INSERT INTO `'.$this->getTableName().'`(`key`, `value`) VALUES(:k, :v)');
 98:         return $sth->execute(array(':k' => $key, ':v' => $value));
 99:     }
100:     
101:     /**
102:      * Get cached data by given key.
103:      * 
104:      * For consistency, the key will go through sha1() 
105:      * before it will be used to retrieve the cached data.
106:      * 
107:      * This method implements {@link MailWizzApi_Cache_Abstract::get()}.
108:      * 
109:      * @param string $key
110:      * @return mixed
111:      */
112:     public function get($key)
113:     {
114:         $key = sha1($key);
115:         
116:         if (isset($this->_loaded[$key])) {
117:             return $this->_loaded[$key];
118:         }
119: 
120:         $con = $this->getConnection();
121:         $sth = $con->prepare('SELECT `value` FROM `'.$this->getTableName().'` WHERE `key` = :k LIMIT 1');
122:         $sth->execute(array(':k' => $key));
123:         $row = $sth->fetch(PDO::FETCH_ASSOC);
124:         return $this->_loaded[$key] = !empty($row['value']) ? unserialize($row['value']) : null;    
125:     }
126:     
127:     /**
128:      * Delete cached data by given key.
129:      * 
130:      * For consistency, the key will go through sha1() 
131:      * before it will be used to delete the cached data.
132:      * 
133:      * This method implements {@link MailWizzApi_Cache_Abstract::delete()}.
134:      * 
135:      * @param string $key
136:      * @return bool
137:      */
138:     public function delete($key)
139:     {
140:         $key = sha1($key);
141:         
142:         if (isset($this->_loaded[$key])) {
143:             unset($this->_loaded[$key]);
144:         }
145:         
146:         $con = $this->getConnection();
147:         $sth = $con->prepare('DELETE FROM `'.$this->getTableName().'` WHERE `key` = :k');
148:         return $sth->execute(array(':k' => $key));
149:     }

307:      * Create the database table if it doesn't exists.
308:      * 
309:      * Please make sure you disable the table creation after the table has been created 
310:      * otherwise it will cause unnecessary overhead.
311:      */
312:     protected function createTable()
313:     {
314:         if ($this->_driver === 'sqlite') {
315:             $sql = '
316:             BEGIN;
317:             CREATE TABLE IF NOT EXISTS `'.$this->getTableName().'` (
318:               `id` INTEGER PRIMARY KEY AUTOINCREMENT,
319:               `key` CHAR(40) NOT NULL,
320:               `value` BLOB NOT NULL
321:             );
322:             CREATE INDEX `key` ON `'.$this->getTableName().'` (`key`);
323:             COMMIT;
324:             ';  
325:         } elseif ($this->_driver === 'mysql') {
326:             $sql = '
327:             CREATE TABLE IF NOT EXISTS `'.$this->getTableName().'` (
328:               `id` INT(11) NOT NULL AUTO_INCREMENT,
329:               `key` CHAR(40) NOT NULL,
330:               `value` LONGBLOB NOT NULL,
331:               PRIMARY KEY (`id`),
332:               KEY `key` (`key`)
333:             ) ENGINE=MyISAM DEFAULT CHARSET=utf8 AUTO_INCREMENT=1 ;
334:             ';
335:         } else {
336:             throw new Exception('The "'.$this->_driver.'" driver is not implemented.');
337:         }
338: 
339:         $this->_connection->exec($sql);
340:     }

342:     /**
343:      * Get the PDO connection.
344:      * 
345:      * @return PDO
346:      */
347:     public function getConnection()
348:     {
349:         if ($this->_connection === null && $this->getImportConnection() instanceof PDO) {
350:             $this->_connection = $this->getImportConnection();
351:         }
352:         
353:         if ($this->_connection instanceof PDO) {
354:             return $this->_connection;
355:         }
356:         
357:         $connectionParts = explode(':', $this->getConnectionString());
358:         $this->_driver = array_shift($connectionParts);
359:         
360:         if ($this->getCreateTable() && $this->_driver === 'sqlite') {
361:             $dbPath = array_shift($connectionParts);
362:             $dir = dirname($dbPath);
363:             if (!is_dir($dir)) {
364:                 throw new Exception('The database storage directory: "'.$dir.'" does not exists.');
365:             }
366:             if (!is_writable($dir)) {
367:                 throw new Exception('The database storage path: "'.$dir.'" is not writable.');
368:             }
369:             if (!is_file($dbPath)) {
370:                 touch($dbPath);
371:             }
372:         } 
373:         
374:         $this->_connection  = new PDO($this->getConnectionString(), $this->getUsername(), $this->getPassword());
375:         $this->_connection->setAttribute(PDO::ATTR_ERRMODE, PDO::ERRMODE_EXCEPTION);
376:         
377:         if ($this->getCreateTable()) {
378:             $this->createTable();
379:         }
380:         
381:         return $this->_connection;
382:     }
