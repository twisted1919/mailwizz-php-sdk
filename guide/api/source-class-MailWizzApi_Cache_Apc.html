12: /**
13:  * MailWizzApi_Cache_Apc makes use of the APC extension in order to cache data in memory.
14:  * 
15:  * As all the data will stay in memory, it is recommeded that will be used only if 
16:  * the system has enough memory, or for development/small servers.
17:  * 
18:  * @author Serban George Cristian <cristian.serban@mailwizz.com>
19:  * @package MailWizzApi
20:  * @subpackage Cache
21:  * @since 1.0
22:  */
23: class MailWizzApi_Cache_Apc extends MailWizzApi_Cache_Abstract
24: {
25:     /**
26:      * Cache data by given key.
27:      * 
28:      * For consistency, the key will go through sha1() before it is saved.
29:      * 
30:      * This method implements {@link MailWizzApi_Cache_Abstract::set()}.
31:      * 
32:      * @param string $key
33:      * @param mixed $value
34:      * @return bool
35:      */
36:     public function set($key, $value)
37:     {
38:         return apc_store(sha1($key), $value, 0);
39:     }
40:     
41:     /**
42:      * Get cached data by given key.
43:      * 
44:      * For consistency, the key will go through sha1() 
45:      * before it will be used to retrieve the cached data.
46:      * 
47:      * This method implements {@link MailWizzApi_Cache_Abstract::get()}.
48:      * 
49:      * @param string $key
50:      * @return mixed
51:      */
52:     public function get($key)
53:     {
54:         return apc_fetch(sha1($key));   
55:     }
56:     
57:     /**
58:      * Delete cached data by given key.
59:      * 
60:      * For consistency, the key will go through sha1() 
61:      * before it will be used to delete the cached data.
62:      * 
63:      * This method implements {@link MailWizzApi_Cache_Abstract::delete()}.
64:      * 
65:      * @param string $key
66:      * @return bool
67:      */
68:     public function delete($key)
69:     {
70:         return apc_delete(sha1($key));
71:     }
72:     
73:     /**
74:      * Delete all cached data.
75:      * 
76:      * This method implements {@link MailWizzApi_Cache_Abstract::flush()}.
77:      * 
78:      * @return bool
79:      */
80:     public function flush()
81:     {
82:         return apc_clear_cache('user');
83:     }
84: }
